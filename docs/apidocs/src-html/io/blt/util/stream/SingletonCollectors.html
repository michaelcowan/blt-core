025package io.blt.util.stream;
026
027import java.util.Optional;
028import java.util.Set;
029import java.util.function.BiConsumer;
030import java.util.function.BinaryOperator;
031import java.util.function.Function;
032import java.util.function.Supplier;
033import java.util.stream.Collector;
034
035import static java.util.Collections.emptySet;
036import static java.util.Objects.nonNull;
037
038/**
039 * Implementations of {@link Collector} that reduce to exactly one or zero elements.
040 * e.g.,
041 * <pre>{@code
042 * var maybeActiveStep = sequentialSteps.stream()
043 *         .filter(SequentialStep::isActive)
044 *         .collect(SingletonCollectors.toOptional());
045 * }</pre>
046 * <p>
047 * If more than one element is present, then {@code IllegalArgumentException} is thrown.
048 */
049public final class SingletonCollectors {
050
051    private SingletonCollectors() {
052    }
053
054    /**
055     * Returns a {@code Collector} that accumulates the only element, if any, into an {@code Optional}.
056     *
057     * @param <T> the type of the input elements
058     * @return a {@code Collector} that accumulates the only element, if any, into an {@code Optional}.
059     * @throws IllegalArgumentException if more than one element is present
060     */
061    public static <T> SingletonCollector<T, Optional<T>> toOptional() {
062        return new SingletonCollector<>(Container::getOptional);
063    }
064
065    /**
066     * Returns a {@code Collector} that accumulates the only element, if any, into a nullable {@code Object}.
067     *
068     * @param <T> the type of the input elements
069     * @return a {@code Collector} that accumulates the only element, if any, into a nullable {@code Object}.
070     * @throws IllegalArgumentException if more than one element is present
071     */
072    public static <T> SingletonCollector<T, T> toNullable() {
073        return new SingletonCollector<>(Container::getValue);
074    }
075
076    private static class SingletonCollector<T, R> implements Collector<T, Container<T>, R> {
077
078        private final Function<Container<T>, R> finisher;
079
080        public SingletonCollector(Function<Container<T>, R> finisher) {
081            this.finisher = finisher;
082        }
083
084        @Override
085        public Supplier<Container<T>> supplier() {
086            return Container::new;
087        }
088
089        @Override
090        public BiConsumer<Container<T>, T> accumulator() {
091            return Container::setValue;
092        }
093
094        @Override
095        public BinaryOperator<Container<T>> combiner() {
096            return null; // Only used in a parallel stream, which this collector doesn't support
097        }
098
099        @Override
100        public Function<Container<T>, R> finisher() {
101            return finisher;
102        }
103
104        @Override
105        public Set<Characteristics> characteristics() {
106            return emptySet();
107        }
108    }
109
110    private static final class Container<T> {
111
112        private T value;
113
114        public T getValue() {
115            return value;
116        }
117
118        public void setValue(T value) {
119            if (nonNull(this.value)) {
120                throw new IllegalStateException("Expected stream to contain exactly 0 or 1 elements");
121            }
122            this.value = value;
123        }
124
125        public Optional<T> getOptional() {
126            return Optional.ofNullable(getValue());
127        }
128    }
129
130}
